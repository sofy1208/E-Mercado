CREATE TABLE `cart` (
	`ID` INT(11) NOT NULL AUTO_INCREMENT,
	`name` VARCHAR(50) NULL DEFAULT '' COLLATE 'latin1_swedish_ci',
	`unitCost` FLOAT NULL DEFAULT NULL,
	`currency` VARCHAR(50) NULL DEFAULT NULL COLLATE 'latin1_swedish_ci',
	`count` INT(11) NULL DEFAULT NULL,
	`images` BLOB NULL DEFAULT NULL,
	PRIMARY KEY (`ID`) USING BTREE
)
COLLATE='latin1_swedish_ci'
ENGINE=InnoDB
AUTO_INCREMENT=5
;
